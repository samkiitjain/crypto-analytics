# Stop and remove all containers, networks
docker-compose down --remove-orphans

# Remove the postgres volume
docker volume rm airflow_postgres-db-volume

# Remove all unused images (this clears the cached layers)
docker image prune -a

# Rebuild from scratch
cd ..
docker build --no-cache -f airflow/Dockerfile -t crypto-airflow:latest .

# Re-init the database
cd airflow
docker-compose up airflow-init

# Start the stack
docker-compose up -d
